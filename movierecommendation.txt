        let lastRecommendedMovie = "";
        function recommendMovie() {
            let userInput = document.getElementById("userInput").value.trim().toLowerCase();
            let genre = document.getElementById("genre").value;
            let matchingMovies = movies[genre].filter(movie => userInput && movie.toLowerCase().includes(userInput));
            
            let recommendedMovie;
            let availableMovies = movies[genre].filter(movie => movie !== lastRecommendedMovie);
            if (matchingMovies.length > 0) {
                recommendedMovie = matchingMovies[Math.floor(Math.random() * matchingMovies.length)];
            } else {
