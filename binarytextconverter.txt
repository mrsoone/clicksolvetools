    <script>
        function convertBinaryToText() {
            const encoding = document.getElementById("encoding").value;
            const binaryInput = document.getElementById("binaryInput").value.trim();
            if (!binaryInput) {
                document.getElementById("output").innerText = "Please enter binary code.";
                return;
            }
            
            try {
                let text = binaryInput.split(/\s+/)
                    .map(bin => {
                        if (!/^[01]{8}$/.test(bin)) throw new Error("Invalid binary format");
                        return String.fromCharCode(parseInt(bin, 2));
                    })
                    .join("");
                
                // ASCII only allows characters in the range 0-127
                if (encoding === "ascii") {
                    text = text.split('').map(char => {
                        return char.charCodeAt(0) < 128 ? char : '?'; // Replace non-ASCII characters with '?'
                    }).join('');
                }
                
                document.getElementById("output").innerText = "Converted Text: " + text;
            } catch (e) {
                document.getElementById("output").innerText = "Invalid binary input. Ensure you use 8-bit binary codes separated by spaces.";
            }
        }
    </script>
