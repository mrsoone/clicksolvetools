    <script>
        let totalWinnings = 0;
        function spinSlots() {
            const symbols = ["🍒", "🍋", "🍉", "⭐", "🔔", "7️⃣"];
            const slot1 = document.getElementById("slot1");
            const slot2 = document.getElementById("slot2");
            const slot3 = document.getElementById("slot3");
            const result = document.getElementById("result");
            const winningsDisplay = document.getElementById("winnings");
            
            slot1.classList.add("spinning");
            slot2.classList.add("spinning");
            slot3.classList.add("spinning");
            
            setTimeout(() => {
                slot1.classList.remove("spinning");
                slot2.classList.remove("spinning");
                slot3.classList.remove("spinning");
                
                let spin1 = symbols[Math.floor(Math.random() * symbols.length)];
                let spin2 = symbols[Math.floor(Math.random() * symbols.length)];
                let spin3 = symbols[Math.floor(Math.random() * symbols.length)];
                
                slot1.innerText = spin1;
                slot2.innerText = spin2;
                slot3.innerText = spin3;
                
                let winnings = 0;
                if (spin1 === spin2 && spin2 === spin3) {
                    winnings = 100;
                    result.innerText = "🎉 JACKPOT! You win $100! 🎉";
                } 
                
                let cherryCount = [spin1, spin2, spin3].filter(symbol => symbol === "🍒").length;
                if (cherryCount > 0) {
                    winnings += cherryCount * 10;
                    result.innerText = `🍒 You got ${cherryCount} cherry(ies)! You win $${cherryCount * 10}!`;
                }
                
                if (winnings === 0) {
                    result.innerText = "Try again!";
                }
                
                totalWinnings += winnings;
                winningsDisplay.innerText = "Winnings: $" + totalWinnings;
            }, 2000);
        }
    </script>
