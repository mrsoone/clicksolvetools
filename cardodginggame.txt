<body>
    <h1>🚗 Car Dodger 🏁</h1>
    <p id="instructions">Use ← → keys or Tap Left/Right side to move!</p>

    <!-- Game Area -->
    <div class="game-container" id="game-container">
        <div id="player-car" class="game-object player-car">🚗</div>
        <!-- Obstacles added dynamically -->

        <!-- Game Over Screen -->
        <div id="game-over-screen">
             <h2>Game Over!</h2>

    <p id="score" style="font-size: clamp(18px, 4vw, 24px); margin-top: 10px;">Score: 0</p>

    <script>
        // --- DOM Elements ---
        const gameContainer = document.getElementById('game-container');
        const playerCarElement = document.getElementById('player-car');
        const scoreDisplay = document.getElementById('score');
        const gameOverScreen = document.getElementById('game-over-screen');
        const finalScoreDisplay = document.getElementById('final-score');
        const newGameBtn = document.getElementById('new-game-btn');
        const instructions = document.getElementById('instructions');

        // --- Game Configuration ---
        const OBSTACLE_TYPES = ["🚙", "🏍️", "🚌", "🚚", "🏎️"];
        const PLAYER_MOVE_SPEED = 0.08; // Fraction of game width per move increment
        const INITIAL_OBSTACLE_SPEED = 0.2; // Units per second (relative to game height)
        const OBSTACLE_SPEED_INCREASE = 0.005; // Speed increase per score point
        const INITIAL_SPAWN_INTERVAL = 1300; // Milliseconds
        const MIN_SPAWN_INTERVAL = 500; // Minimum spawn time
        const SPAWN_RATE_INCREASE = 10; // Milliseconds reduction per score point
        const PLAYER_BOTTOM_MARGIN = 10; // Px from bottom
        const COLLISION_PADDING = 0.15; // % padding for collision box (makes it slightly smaller than visual)

        // --- Game State Variables ---
        let score = 0;
        let playerLane = 2; // 0=leftmost, 4=rightmost (5 lanes total conceptually)
        let targetPlayerX = 0; // Target X position for smooth movement
        let currentObstacleSpeed = INITIAL_OBSTACLE_SPEED;
        let currentSpawnInterval = INITIAL_SPAWN_INTERVAL;
        let obstacles = []; // Array to hold { element: DOMElement, y: number } objects
        let gameOver = false;
        let lastTimestamp = 0;
        let timeSinceLastSpawn = 0;
        let animationFrameId = null;
        let gameWidth = 0;
        let gameHeight = 0;
        let carWidth = 0;
        let carHeight = 0;
        let laneWidth = 0;
        let lanePositions = [];

        // --- Helper Functions ---
        function pxToNumber(pxString) {
            return parseFloat(pxString) || 0;
        }

        function updateGameDimensions() {
            const rect = gameContainer.getBoundingClientRect();
            gameWidth = rect.width;
            gameHeight = rect.height;
            // Calculate dynamic sizes based on container
            carWidth = gameWidth * parseFloat(getComputedStyle(document.documentElement).getPropertyValue('--car-width-ratio'));
            carHeight = gameHeight * parseFloat(getComputedStyle(document.documentElement).getPropertyValue('--car-height-ratio'));
             // Ensure max sizes are respected
             const maxCarWidth = pxToNumber(getComputedStyle(playerCarElement).maxWidth);
             const maxCarHeight = pxToNumber(getComputedStyle(playerCarElement).maxHeight);
             carWidth = Math.min(carWidth, maxCarWidth);
             carHeight = Math.min(carHeight, maxCarHeight);

             // Update CSS Variables if needed dynamically (or rely on initial CSS calc)
             // playerCarElement.style.width = `${carWidth}px`;
             // playerCarElement.style.height = `${carHeight}px`;


            laneWidth = gameWidth / 5; // Assuming 5 lanes
            lanePositions = Array.from({ length: 5 }, (_, i) => (i * laneWidth) + (laneWidth / 2) - (carWidth / 2));

             // Set initial player position
             targetPlayerX = lanePositions[playerLane];
             playerCarElement.style.transform = `translateX(${targetPlayerX}px)`;
             playerCarElement.style.bottom = `${PLAYER_BOTTOM_MARGIN}px`; // Set bottom margin once

            console.log(`Game Dimensions Updated: ${gameWidth}x${gameHeight}, Car: ${carWidth}x${carHeight}`);
        }


        function getRandomLaneIndex() {
            return Math.floor(Math.random() * lanePositions.length);
        }

        // --- Input Handling ---
        function movePlayer(direction) { // direction is -1 for left, 1 for right
            if (gameOver) return;
            let newLane = playerLane + direction;
            // Clamp lane index between 0 and 4
            newLane = Math.max(0, Math.min(4, newLane));
            if (newLane !== playerLane) {
                playerLane = newLane;
                targetPlayerX = lanePositions[playerLane]; // Set target, movement happens in game loop
            }
        }

        function handleKeyDown(event) {
            if (event.key === "ArrowLeft") {
                movePlayer(-1);
            } else if (event.key === "ArrowRight") {
                movePlayer(1);
            } else if (event.key === " " && gameOver) { // Space to restart
                resetGame();
            }
        }

        function handleTouch(event) {
            if (gameOver) return;
            // Prevent scrolling/zooming
            event.preventDefault();
            const touchX = event.touches[0].clientX;
            const gameRect = gameContainer.getBoundingClientRect();
            const touchXRelative = touchX - gameRect.left;

            if (touchXRelative < gameWidth / 2) {
                movePlayer(-1); // Touched left half
            } else {
                movePlayer(1); // Touched right half
            }
        }

        // --- Game Logic ---
        function createObstacle() {
            const element = document.createElement("div");
            element.classList.add("game-object", "obstacle");
            element.innerText = OBSTACLE_TYPES[Math.floor(Math.random() * OBSTACLE_TYPES.length)];

            const laneIndex = getRandomLaneIndex();
            const startX = lanePositions[laneIndex];
            const startY = -carHeight; // Start just above the screen

            element.style.transform = `translate(${startX}px, ${startY}px)`;

            gameContainer.appendChild(element);
            obstacles.push({ element, x: startX, y: startY, lane: laneIndex });
        }

        function updateObstacles(deltaTime) {
             const speedPxPerMs = currentObstacleSpeed * gameHeight / 1000; // Convert relative speed to px/ms

             obstacles.forEach((obstacle, index) => {
                obstacle.y += speedPxPerMs * deltaTime;
                obstacle.element.style.transform = `translate(${obstacle.x}px, ${obstacle.y}px)`;

                // Remove obstacle if it goes off screen
                if (obstacle.y > gameHeight) {
                    gameContainer.removeChild(obstacle.element);
                    obstacles.splice(index, 1);
                    incrementScore();
                }
            });
        }

         function updatePlayer(deltaTime) {
            // Smoothly move player towards target X
            const currentX = parseFloat(playerCarElement.style.transform.match(/translateX\(([-0-9.]+)/)?.[1]) || targetPlayerX;
            const dx = targetPlayerX - currentX;

            // Adjust movement speed based on distance to target for a smoother feel
             const moveAmount = dx * 0.2; // Move 20% of the distance each frame

             // Prevent overshooting
            const newX = Math.abs(dx) < 1 ? targetPlayerX : currentX + moveAmount;

            playerCarElement.style.transform = `translateX(${newX}px)`;
         }


        function checkCollisions() {
            const playerY = gameHeight - carHeight - PLAYER_BOTTOM_MARGIN; // Player's effective Y
            const playerX = parseFloat(playerCarElement.style.transform.match(/translateX\(([-0-9.]+)/)?.[1]) || targetPlayerX;

            // Simplified collision boxes (adjust padding as needed)
            const playerLeft = playerX + carWidth * COLLISION_PADDING;
            const playerRight = playerX + carWidth * (1 - COLLISION_PADDING);
            const playerTop = playerY + carHeight * COLLISION_PADDING;
            const playerBottom = playerY + carHeight * (1 - COLLISION_PADDING);


            for (const obstacle of obstacles) {
                const obsLeft = obstacle.x + carWidth * COLLISION_PADDING;
                const obsRight = obstacle.x + carWidth * (1 - COLLISION_PADDING);
                const obsTop = obstacle.y + carHeight * COLLISION_PADDING;
                const obsBottom = obstacle.y + carHeight * (1 - COLLISION_PADDING);

                // Basic AABB collision check
                if (
                    playerLeft < obsRight &&
                    playerRight > obsLeft &&
                    playerTop < obsBottom &&
                    playerBottom > obsTop
                ) {
                     // Collision detected!
                     showExplosion(playerX + carWidth / 2, playerY + carHeight / 2);
                     triggerGameOver();
                     return; // Exit check early
                }
            }
        }

        function showExplosion(x, y) {
            const explosion = document.createElement("div");
            explosion.classList.add("explosion");
            explosion.innerText = "💥";
            // Position relative to the container, centered on collision point
            explosion.style.left = `${x - carWidth * 0.5}px`; // Adjust based on explosion font size
            explosion.style.top = `${y - carHeight * 0.5}px`; // Adjust based on explosion font size

            gameContainer.appendChild(explosion);
            setTimeout(() => {
                if (explosion.parentElement === gameContainer) {
                    gameContainer.removeChild(explosion);
                }
            }, 400); // Match animation duration
        }

        function incrementScore() {
            score++;
            scoreDisplay.innerText = "Score: " + score;
            // Increase difficulty
            currentObstacleSpeed += OBSTACLE_SPEED_INCREASE;
            currentSpawnInterval = Math.max(MIN_SPAWN_INTERVAL, currentSpawnInterval - SPAWN_RATE_INCREASE);
        }

        function triggerGameOver() {
            if (gameOver) return; // Prevent multiple triggers
            gameOver = true;
            cancelAnimationFrame(animationFrameId); // Stop the game loop
            console.log("Game Over! Final Score:", score);

            // Display game over screen
            finalScoreDisplay.innerText = "Final Score: " + score;
            gameOverScreen.style.display = "flex"; // Show the screen

            // Optional: Add a slight delay before showing button focus if needed
            setTimeout(() => newGameBtn.focus(), 100);
             instructions.style.display = 'none'; // Hide instructions
        }

        // --- Game Loop ---
        function gameLoop(timestamp) {
            if (gameOver) return;

            const deltaTime = timestamp - lastTimestamp;
            lastTimestamp = timestamp;
            timeSinceLastSpawn += deltaTime;

            // Update game elements
             updatePlayer(deltaTime); // Handle player movement smoothing
            updateObstacles(deltaTime);

            // Spawn new obstacles
            if (timeSinceLastSpawn > currentSpawnInterval) {
                createObstacle();
                timeSinceLastSpawn = 0; // Reset spawn timer
            }

            // Check for collisions
            checkCollisions();

            // Request the next frame
            animationFrameId = requestAnimationFrame(gameLoop);
        }

        // --- Game Initialization and Reset ---
        function resetGame() {
            console.log("Resetting game...");
            // Reset state variables
            score = 0;
            playerLane = 2; // Center lane
            obstacles.forEach(obstacle => gameContainer.removeChild(obstacle.element));
            obstacles = [];
            currentObstacleSpeed = INITIAL_OBSTACLE_SPEED;
            currentSpawnInterval = INITIAL_SPAWN_INTERVAL;
            gameOver = false;
            timeSinceLastSpawn = 0;
            lastTimestamp = performance.now(); // Reset timestamp base

            // Reset UI
            scoreDisplay.innerText = "Score: 0";
            gameOverScreen.style.display = "none"; // Hide game over screen
             instructions.style.display = 'block'; // Show instructions

            // Recalculate dimensions and reposition player
            updateGameDimensions();


            // Cancel any existing animation frame just in case
            if (animationFrameId) {
                cancelAnimationFrame(animationFrameId);
            }

            // Start the game loop again
            animationFrameId = requestAnimationFrame(gameLoop);
        }

        function setupEventListeners() {
             document.addEventListener("keydown", handleKeyDown);
             // Use touchstart for mobile responsiveness - prevents delay
             gameContainer.addEventListener("touchstart", handleTouch, { passive: false }); // passive:false to allow preventDefault
             newGameBtn.addEventListener("click", resetGame);
              // Update dimensions on resize/orientation change
             window.addEventListener('resize', updateGameDimensions);
             window.addEventListener('orientationchange', updateGameDimensions);
        }

        // --- Start Game ---
        function startGame() {
            console.log("Starting game...");
             setupEventListeners();
             resetGame(); // Initial setup and start
        }

        // Wait for the DOM to be fully loaded
        document.addEventListener('DOMContentLoaded', startGame);

    </script>
