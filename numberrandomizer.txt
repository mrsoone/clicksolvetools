        <input type="number" id="maxValue" placeholder="Max">
        <br>
        <button onclick="generateRandomNumber()">Generate</button>
        <p id="result">Click the button to get a number!</p>
    </div>

    <script>
        function generateRandomNumber() {
            let min = parseInt(document.getElementById("minValue").value);
            let max = parseInt(document.getElementById("maxValue").value);
            
            if (isNaN(min) || isNaN(max) || min >= max) {
                document.getElementById("result").textContent = "Please enter a valid range!";
                return;
            }
            
            let randomNumber = Math.floor(Math.random() * (max - min + 1)) + min;
            document.getElementById("result").textContent = "Random Number: " + randomNumber;
        }
    </script>
