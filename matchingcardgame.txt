<body>
    <h1>🃏 Card Matching Memory Game 🃏</h1>
    <div class="game-board" id="game-board"></div>
    <p id="result">Find all the matching pairs!</p>
    <button onclick="resetGame()">Restart Game</button>
    
    <script>
        const symbols = ["🍎", "🍎", "🍌", "🍌", "🍒", "🍒", "🍇", "🍇", "🥑", "🥑", "🍉", "🍉", "🍍", "🍍", "🥕", "🥕"];
        let shuffledSymbols = shuffle(symbols);
        let firstCard = null;
        let secondCard = null;
        let lockBoard = false;
        let matchedPairs = 0;

        function shuffle(array) {
            return array.sort(() => 0.5 - Math.random());
        }

        function createBoard() {
            const gameBoard = document.getElementById("game-board");
            gameBoard.innerHTML = "";
            shuffledSymbols.forEach((symbol, index) => {
                const card = document.createElement("div");
                card.classList.add("card");
                card.dataset.symbol = symbol;
                card.dataset.index = index;
                card.addEventListener("click", flipCard);
                gameBoard.appendChild(card);
            });
        }

        function flipCard() {
            if (lockBoard || this.classList.contains("flipped")) return;
            
            this.innerText = this.dataset.symbol;
            this.classList.add("flipped");
            
            if (!firstCard) {
                firstCard = this;
                return;
            }
            
            secondCard = this;
            lockBoard = true;
            checkMatch();
        }

        function checkMatch() {
            if (firstCard.dataset.symbol === secondCard.dataset.symbol) {
                matchedPairs++;
                resetTurn();
                if (matchedPairs === symbols.length / 2) {
                    document.getElementById("result").innerText = "🎉 You matched all pairs! 🎉";
                }
            } else {
                setTimeout(() => {
                    firstCard.innerText = "";
                    secondCard.innerText = "";
                    firstCard.classList.remove("flipped");
                    secondCard.classList.remove("flipped");
                    resetTurn();
                }, 800);
            }
        }

        function resetTurn() {
            firstCard = null;
            secondCard = null;
            lockBoard = false;
        }

        function resetGame() {
            matchedPairs = 0;
            shuffledSymbols = shuffle(symbols);
            createBoard();
            document.getElementById("result").innerText = "Find all the matching pairs!";
        }

        createBoard();
    </script>
