    <div class="container">
        <h2>Word & Character Counter</h2>
        <textarea id="textInput" placeholder="Type or paste your text here..." oninput="countWordsAndChars()"></textarea>
        <p class="info">Words: <span id="wordCount">0</span></p>
        <p class="info">Characters: <span id="charCount">0</span></p>
    </div>
    <script>
        function countWordsAndChars() {
            let text = document.getElementById("textInput").value;
            let charCount = text.length;
            let wordCount = text.trim() === "" ? 0 : text.trim().split(/\s+/).length;
            
            document.getElementById("wordCount").textContent = wordCount;
            document.getElementById("charCount").textContent = charCount;
        }
    </script>
