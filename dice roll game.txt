    <button onclick="rollDice()">Roll Dice</button>
    <p id="result">Result: ?</p>

    <script>
        function rollDice() {
            let dice1 = document.getElementById("dice1");
            let dice2 = document.getElementById("dice2");
            let result = document.getElementById("result");
            
            // Add rolling animation
            dice1.classList.add("rolling");
            dice2.classList.add("rolling");
            result.innerText = "Rolling...";
            
            setTimeout(() => {
                dice1.classList.remove("rolling");
                dice2.classList.remove("rolling");
                let diceValue1 = Math.floor(Math.random() * 6) + 1;
                let diceValue2 = Math.floor(Math.random() * 6) + 1;
                let diceFaces = ["⚀", "⚁", "⚂", "⚃", "⚄", "⚅"];
                dice1.innerText = diceFaces[diceValue1 - 1];
                dice2.innerText = diceFaces[diceValue2 - 1];
                result.innerText = `Result: ${diceValue1} + ${diceValue2} = ${diceValue1 + diceValue2}`;
            }, 1000);
        }
    </script>
