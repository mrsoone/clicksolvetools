    <script>
        function createRuler() {
            const ruler = document.getElementById("ruler");
            
            for (let i = 0; i <= 12; i++) {
                let tick = document.createElement("div");
                tick.className = "tick";
                tick.style.height = i % 2 === 0 ? "30px" : "20px";
                tick.style.left = (i / 12) * 100 + "%";
                ruler.appendChild(tick);
                
                let label = document.createElement("div");
                label.className = "tick-label";
                label.style.left = (i / 12) * 100 + "%";
                label.innerText = i;
                ruler.appendChild(label);
            }
            
            for (let i = 0; i <= 30; i++) {
                let cmTick = document.createElement("div");
                cmTick.className = "tick";
                cmTick.style.height = i % 5 === 0 ? "25px" : "15px";
                cmTick.style.left = (i / 30) * 100 + "%";
                cmTick.style.background = "#1976d2";
                ruler.appendChild(cmTick);
                
                if (i % 5 === 0) {
                    let cmLabel = document.createElement("div");
                    cmLabel.className = "cm-label";
                    cmLabel.style.left = (i / 30) * 100 + "%";
                    cmLabel.innerText = i;
                    ruler.appendChild(cmLabel);
                }
            }
            
            for (let i = 0; i <= 300; i++) {
                let mmTick = document.createElement("div");
                mmTick.className = "mm-tick";
                mmTick.style.left = (i / 300) * 100 + "%";
                ruler.appendChild(mmTick);
                
                if (i % 10 === 0) {
                    let mmLabel = document.createElement("div");
                    mmLabel.className = "mm-label";
                    mmLabel.style.left = (i / 300) * 100 + "%";
                    mmLabel.innerText = i;
                    ruler.appendChild(mmLabel);
                }
            }
        }
        createRuler();
    </script>
