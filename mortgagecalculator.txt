    <script>
        document.addEventListener('DOMContentLoaded', function() {

            // Cache DOM elements
            const loanAmountInput = document.getElementById('loanAmount');
            const interestRateInput = document.getElementById('interestRate');
            const loanTermSelect = document.getElementById('loanTerm');
            const customTermInput = document.getElementById('customTerm');
            const propertyTaxInput = document.getElementById('propertyTax');
            const homeInsuranceInput = document.getElementById('homeInsurance'); // Added
            const pmiInput = document.getElementById('pmi');
            const hoaFeesInput = document.getElementById('hoaFees'); // Added
            const resultText = document.getElementById('resultText');

            // --- Attach Event Listeners ---
             // Add new inputs to the list
            const inputs = [loanAmountInput, interestRateInput, loanTermSelect, customTermInput, propertyTaxInput, homeInsuranceInput, pmiInput, hoaFeesInput];
            inputs.forEach(input => {
                input.addEventListener('input', calculateMortgage);
                 if (input.tagName === 'SELECT') {
                    input.removeEventListener('input', calculateMortgage);
                    input.addEventListener('change', calculateMortgage);
                 }
            });

            // Listener to show/hide custom term input
            loanTermSelect.addEventListener('change', function() {
                const isCustom = this.value === 'custom';
                customTermInput.style.display = isCustom ? 'block' : 'none';
                if (isCustom) {
                    customTermInput.value = '';
                    customTermInput.focus();
                    resultText.textContent = 'Enter custom loan term in years.';
                    resultText.classList.remove('error');
                } else {
                    calculateMortgage();
                }
            });

            // --- Calculation Function ---
            function calculateMortgage() {
                // Get and parse values
                const P = parseFloat(loanAmountInput.value) || 0;
                const annualRate = parseFloat(interestRateInput.value) || 0;
                const termSelection = loanTermSelect.value;
                const annualTax = parseFloat(propertyTaxInput.value) || 0;
                const annualInsurance = parseFloat(homeInsuranceInput.value) || 0; // Added
                const annualPmi = parseFloat(pmiInput.value) || 0;
                const monthlyHoa = parseFloat(hoaFeesInput.value) || 0; // Added (already monthly)

                let termYears;
                if (termSelection === 'custom') {
                    termYears = parseFloat(customTermInput.value) || 0;
                     if (termYears <= 0 || termYears > 50) {
                        showError('Please enter a valid custom term (1-50 years).');
                         return;
                     }
                } else {
                    termYears = parseInt(termSelection);
                }

                // --- Input Validation ---
                if (P <= 0) { showError('Please enter a valid Loan Amount.'); return; }
                if (annualRate < 0) { showError('Please enter a valid Interest Rate.'); return; }
                if (termYears <= 0) { showError('Please select or enter a valid Loan Term.'); return; }
                 // Optional fields validation (ensure non-negative)
                 if (annualTax < 0) { showError('Annual Property Tax cannot be negative.'); return; }
                 if (annualInsurance < 0) { showError('Annual Homeowners Insurance cannot be negative.'); return; }
                 if (annualPmi < 0) { showError('Annual PMI cannot be negative.'); return; }
                 if (monthlyHoa < 0) { showError('Monthly HOA Fees cannot be negative.'); return; }


                // --- Calculations ---
                const i = annualRate / 100 / 12;
                const n = termYears * 12;

                let M; // Monthly Principal & Interest
                if (i === 0) { M = P / n; }
                else {
                     const numerator = P * i * Math.pow(1 + i, n);
                     const denominator = Math.pow(1 + i, n) - 1;
                     M = numerator / denominator;
                 }

                 if (!isFinite(M)) {
                     showError('Calculation resulted in an invalid number. Check inputs.');
                     return;
                 }

                const monthlyTax = annualTax / 12;
                const monthlyInsurance = annualInsurance / 12; // Calculate monthly insurance
                const monthlyPmi = annualPmi / 12;
                // monthlyHoa is already monthly

                const totalMonthlyPayment = M + monthlyTax + monthlyInsurance + monthlyPmi + monthlyHoa; // Add new components

                // --- Display Results ---
                displayResults(M, monthlyTax, monthlyInsurance, monthlyPmi, monthlyHoa, totalMonthlyPayment); // Pass new values
            }

            // --- Helper to Display Results ---
            // Updated function signature and content
            function displayResults(pAndI, tax, insurance, pmi, hoa, total) {
                const formatter = new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' });

                resultText.innerHTML = `
                    <strong>Principal & Interest:</strong> ${formatter.format(pAndI)} <br>
                    <strong>Property Tax (Monthly):</strong> ${formatter.format(tax)} <br>
                    <strong>Home Insurance (Monthly):</strong> ${formatter.format(insurance)} <br>
                    <strong>PMI (Monthly):</strong> ${formatter.format(pmi)} <br>
                    <strong>HOA Fees (Monthly):</strong> ${formatter.format(hoa)} <br>
                    <div class="total-payment">
                        <strong>Total Monthly Payment:</strong> ${formatter.format(total)}
                    </div>
                `;
                resultText.classList.remove('error');
            }

             // --- Helper to Show Error Messages ---
            function showError(message) {
                 resultText.textContent = message;
                 resultText.classList.add('error');
             }

        }); // End DOMContentLoaded
    </script>
