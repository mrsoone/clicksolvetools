<body>
    <h1>🔥 Click the Button Game 🔥</h1>
    <p>Click the button as many times as you can in 10 seconds!</p>
    <button id="clickButton" onclick="increaseScore()">💥 Click Me! 💥</button>
    <p id="score">Score: 0</p>
    <p id="timer">Time Left: 10s</p>
    <br>
    <button onclick="startGame()">🎮 Start Game 🎮</button>

    <script>
        let score = 0;
        let timeLeft = 10;
        let gameActive = false;

        function startGame() {
            if (gameActive) return;
            gameActive = true;
            score = 0;
            timeLeft = 10;
            document.getElementById("score").innerText = "Score: 0";
            document.getElementById("timer").innerText = "Time Left: 10s";
            document.body.style.backgroundColor = "#006400";

            let countdown = setInterval(() => {
                timeLeft--;
                document.getElementById("timer").innerText = `Time Left: ${timeLeft}s`;

                if (timeLeft <= 0) {
                    clearInterval(countdown);
                    gameActive = false;
                    document.body.style.backgroundColor = "#282c34";
                    alert(`🎉 Time's up! Your final score is: ${score}`);
                }
            }, 1000);
        }

        function increaseScore() {
            if (gameActive) {
                score++;
                document.getElementById("score").innerText = `Score: ${score}`;
                showScoreEffect();
            }
        }

        function showScoreEffect() {
            let popup = document.createElement("div");
            popup.classList.add("score-popup");
            popup.innerText = "+1";
            document.body.appendChild(popup);

            let x = Math.random() * window.innerWidth * 0.8;
            let y = Math.random() * window.innerHeight * 0.8;
            popup.style.left = `${x}px`;
            popup.style.top = `${y}px`;

            setTimeout(() => { popup.remove(); }, 1000);
        }
    </script>
