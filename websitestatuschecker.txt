    <script>
        async function checkWebsite() {
            const urlInput = document.getElementById("url").value.trim();
            const resultDiv = document.getElementById("result");

            if (!urlInput) {
                resultDiv.innerHTML = "<p style='color: red;'>Please enter a website URL.</p>";
                return;
            }

            resultDiv.innerHTML = "<p>Checking...</p>";

            let url = urlInput.startsWith("http") ? urlInput : `https://${urlInput}`;

            try {
                // Using AllOrigins API to bypass CORS and check if the website is reachable
                const response = await fetch(`https://api.allorigins.win/get?url=${encodeURIComponent(url)}`);

                if (!response.ok) throw new Error("Network error");

                const data = await response.json();

                if (data.contents) {
                    resultDiv.innerHTML = `<p class="up">✅ ${url} is UP! (Successfully Fetched)</p>`;
                } else {
                    resultDiv.innerHTML = `<p class="down">❌ ${url} is DOWN! (No Response)</p>`;
                }
            } catch (error) {
                resultDiv.innerHTML = `<p class="down">❌ ${url} is DOWN (Error: Unable to reach website)</p>`;
            }
        }

        // Event Listener to trigger check when Enter is pressed
        document.addEventListener("DOMContentLoaded", function() {
            document.getElementById("url").addEventListener("keypress", function(event) {
                if (event.keyCode === 13) {  // 13 = Enter key
                    event.preventDefault();
                    checkWebsite();
                }
            });
        });
    </script>

<body>
    <h2>Website Status Checker</h2>
    <input type="text" id="url" placeholder="Enter website URL">
    <button onclick="checkWebsite()">Check</button>

</body>
