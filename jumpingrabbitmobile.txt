        canvas {
            background-color: #add8e6;
            display: block;
            margin: 10px auto;
            border: 2px solid black;
            position: relative;
            max-width: 100vw;
            height: 80vh;
        }
        #message, #scoreDisplay {
            font-size: 20px;
            font-weight: bold;
            margin: 5px;
        }
        @media (max-width: 768px) {
            #message, #scoreDisplay {
                font-size: 16px;
            }

                height: 70vh;
            }

<body>
    <h1>🐇 The Jumping Rabbit 🐇</h1>
    <div id="message">Tap Anywhere to Start</div>
    <div id="scoreDisplay">Score: 0</div>
    <canvas id="gameCanvas"></canvas>
    
    <script>
        const canvas = document.getElementById("gameCanvas");
        const ctx = canvas.getContext("2d");
        canvas.width = window.innerWidth * 0.9;
        canvas.height = window.innerHeight * 0.7;
        
        const messageDisplay = document.getElementById("message");
        const scoreDisplay = document.getElementById("scoreDisplay");

        let rabbit = { x: 50, y: canvas.height / 2 - 15, width: 30, height: 30, velocity: 0, gravity: 0.4, lift: -7, frozen: true };
        let carrots = [];
        let score = 0;
        let gameRunning = false;
        let gameOverState = false;
        let firstStart = true;

        function drawRabbit() {
            ctx.font = "30px Arial";
            ctx.fillText("🐰", rabbit.x, rabbit.y + 25);
        }

        function updateRabbit() {
            if (rabbit.frozen) return;
            rabbit.velocity += rabbit.gravity;
            rabbit.y += rabbit.velocity;
            if (rabbit.y + rabbit.height > canvas.height) gameOver();
            if (rabbit.y < 0) rabbit.y = 0;
        }

        function drawCarrots() {
            carrots.forEach(carrot => {
                ctx.font = "30px Arial";
                ctx.fillText(carrot.symbol, carrot.x, carrot.topHeight);
                ctx.fillText(carrot.symbol, carrot.x, carrot.topHeight + carrot.gap);
            });
        }

        function updateCarrots() {
            carrots.forEach(carrot => {
                carrot.x -= 2;
                if (carrot.x + 50 < rabbit.x && !carrot.passed) {
                    carrot.passed = true;
                    score += 10;
                    scoreDisplay.innerText = "Score: " + score;
                }
                if (checkCollision(carrot)) gameOver();
            });
        }

        function checkCollision(carrot) {
            return (
                rabbit.x < carrot.x + 50 &&
                rabbit.x + rabbit.width > carrot.x &&
                (rabbit.y < carrot.topHeight || rabbit.y + rabbit.height > carrot.topHeight + carrot.gap)
            );
        }

        function addCarrot() {
            let topHeight = Math.floor(Math.random() * (canvas.height / 2));
            let gapSize = Math.max(100 - Math.floor(score / 5) * 5, 60);
            carrots.push({ x: canvas.width, topHeight: topHeight, gap: gapSize, passed: false, symbol: "🥕" });
        }

        function gameOver() {
            gameRunning = false;
            gameOverState = true;
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.font = "30px Arial";
            ctx.fillStyle = "red";
            ctx.fillText("Game Over!", canvas.width / 2 - 80, canvas.height / 2 - 50);
            ctx.fillStyle = "black";
            ctx.fillText("Score: " + score, canvas.width / 2 - 50, canvas.height / 2);
            messageDisplay.innerText = "Tap to Start Again";
            messageDisplay.style.display = "block";
            firstStart = false;
        }

        function startGame() {
            if (gameOverState || firstStart) {
                resetGame();
                firstStart = false;
            }
            rabbit.frozen = false;
            gameRunning = true;
            messageDisplay.style.display = "none";
            gameLoop();
        }

        function resetGame() {
            rabbit.y = canvas.height / 2 - 15;
            rabbit.velocity = 0;
            carrots = [];
            gameOverState = false;
            score = 0;
            scoreDisplay.innerText = "Score: 0";
        }

        function jump() {
            if (!gameRunning) {
                startGame();
            } else {
                rabbit.velocity = rabbit.lift;
            }
        }

        canvas.addEventListener("click", jump);
        document.addEventListener("keydown", jump);
        setInterval(addCarrot, 2000);
        
        function gameLoop() {
            if (!gameRunning) return;
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            updateRabbit();
            updateCarrots();
            drawRabbit();
            drawCarrots();
            requestAnimationFrame(gameLoop);
        }
    </script>
</body>
</html>
