    <script>
        function compareText() {
            let text1 = document.getElementById("text1").value;
            let text2 = document.getElementById("text2").value;
            let diffOutput = document.getElementById("diffOutput");

            let diff = generateDiff(text1, text2);
            diffOutput.innerHTML = diff || "No differences found.";
        }

        function generateDiff(text1, text2) {
            let lines1 = text1.split('\n');
            let lines2 = text2.split('\n');
            let removedText = "<div class='section'><strong>Removed from Box 1:</strong><br>";
            let addedText = "<div class='section'><strong>Added in Box 2:</strong><br>";
            let unchangedText = "<div class='section'><strong>Unchanged (Same Line in Both Boxes):</strong><br>";
            let duplicateText = "<div class='section'><strong>Duplicates (Found on Different Lines):</strong><br>";

            let seenDuplicates = new Set();

            for (let i = 0; i < Math.max(lines1.length, lines2.length); i++) {
                let line1 = lines1[i] || "";
                let line2 = lines2[i] || "";

                if (line1 === line2) {
                    unchangedText += `<div class='same'>✔ ${line1}</div>`;
                } else {
                    if (lines1.includes(line2) && !seenDuplicates.has(line2)) {
                        duplicateText += `<div class='duplicate'>🔁 ${line2}</div>`;
                        seenDuplicates.add(line2);
                    }
                    if (lines2.includes(line1) && !seenDuplicates.has(line1)) {
                        duplicateText += `<div class='duplicate'>🔁 ${line1}</div>`;
                        seenDuplicates.add(line1);
                    }
                    if (line1 && !lines2.includes(line1)) {
                        removedText += `<div class='removed'>❌ ${line1}</div>`;
                    }
                    if (line2 && !lines1.includes(line2)) {
                        addedText += `<div class='added'>✅ ${line2}</div>`;
                    }
                }
            }

            removedText += "</div>";
            addedText += "</div>";
            unchangedText += "</div>";
            duplicateText += "</div>";

            return removedText + addedText + unchangedText + duplicateText;
        }
    </script>
