<body>
    <h2>🎡 Spin the Wheel 🎡</h2>
    <div class="input-container">
        <input type="text" id="options" placeholder="Enter options separate w commas" oninput="setupWheel()">
        <br><br>
        <button onclick="spinWheel()">Spin 🎯</button>
    </div>
    <div class="wheel-container">
        <div class="marker"></div>
        <canvas id="wheelCanvas"></canvas>
    </div>
    
    <script>
        let canvas = document.getElementById("wheelCanvas");
        let ctx = canvas.getContext("2d");
        let wheel = { angle: 0, spinning: false, options: [], colors: [] };
        canvas.width = 400;
        canvas.height = 400;

        function getRandomColors(num) {
            let colors = [];
            for (let i = 0; i < num; i++) {
                colors.push(`hsl(${(i * 360) / num}, 100%, 65%)`);
            }
            return colors;
        }

        function drawWheel(options, colors) {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            let numOptions = options.length;
            let anglePerOption = (2 * Math.PI) / numOptions;
            
            for (let i = 0; i < numOptions; i++) {
                ctx.beginPath();
                ctx.moveTo(200, 200);
                ctx.arc(200, 200, 200, i * anglePerOption, (i + 1) * anglePerOption);
                ctx.closePath();
                ctx.fillStyle = colors[i];
                ctx.fill();
                
                ctx.save();
                ctx.translate(200, 200);
                ctx.rotate(i * anglePerOption + anglePerOption / 2);
                ctx.fillStyle = "black";
                ctx.font = "16px Arial";
                ctx.textAlign = "center";
                ctx.textBaseline = "middle";
                ctx.fillText(options[i], 100, 0);
                ctx.restore();
            }
        }

        function setupWheel() {
            let input = document.getElementById("options").value;
            wheel.options = input.split(",").map(option => option.trim()).filter(option => option !== "");
            if (wheel.options.length === 0) {
                wheel.options = ["Option 1", "Option 2", "Option 3", "Option 4"];
            }
            wheel.colors = getRandomColors(wheel.options.length);
            drawWheel(wheel.options, wheel.colors);
        }

        function spinWheel() {
            if (wheel.options.length === 0) {
                alert("Please enter options first!");
                return;
            }
            let randomAngle = 360 * 3 + Math.random() * 360;
            let rotation = randomAngle * (Math.PI / 180);
            let spinTime = 3000;
            
            wheel.spinning = true;
            let start = performance.now();
            function animate(time) {
                let progress = (time - start) / spinTime;
                if (progress < 1) {
                    wheel.angle = rotation * progress;
                    ctx.setTransform(1, 0, 0, 1, 0, 0);
                    ctx.translate(200, 200);
                    ctx.rotate(wheel.angle);
                    ctx.translate(-200, -200);
                    drawWheel(wheel.options, wheel.colors);
                    requestAnimationFrame(animate);
                } else {
                    let selectedIndex = Math.floor(((wheel.angle + (Math.PI / wheel.options.length)) / (2 * Math.PI)) * wheel.options.length) % wheel.options.length;
                    selectedIndex = (selectedIndex + wheel.options.length) % wheel.options.length;
                    alert("Selected: " + wheel.options[selectedIndex]);
                    wheel.spinning = false;
                }
            }
            requestAnimationFrame(animate);
        }

        setupWheel();
    </script>
